#include <stdio.h>

//增加节点
void add(List *list,int number){
		//建一个新节点
		Node *p = (Node *)malloc(sizeof(Node));
		p->value = number;
		p->next = NULL;
		Node *last = list->head;
		if(head=NULL){
			//从头结点开始找到最后的节点
			while(last->next){
				last = last->next;
			}
			//将最后的节点指向新建的这个节点
			last->next = p;				
		}else{
			list->head = p;
		}
}

//遍历节点
void print(List *list){
	Node *p;
	//经典的链表操作语句
	for(p=list->head;p;p=p->next){
		printf("%d",p->value);
	}
	printf("\n");
}

//寻找节点
Node * find(List* list,int number){
	Node *p;
	for(p=list->head;p;p=p->next){
		if(p->value == number){
			printf("找到了");
			return p;
		}
	}
}

//删除节点

		if(p->value == number){

				list->head = p->next;
			}
			free(p);
			break;
		}
	}
}

//清除链表
void clear(List *list){
	Node *q;
	for(p = list->head;p;p=q){
		q = p->next;
		free(p);
	}
}
